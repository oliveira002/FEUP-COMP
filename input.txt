import io;

class Fac {
    public int computeFac (int num){
        String num_aux;
        if (num < 1)
            num_aux = 23;
        else
            num_aux = num * (this.computeFac(num - 1));
        return num_aux;
    }

    public static void main (String[] args ){
        io.println(new Fac().computeFac(10));
    }
}
